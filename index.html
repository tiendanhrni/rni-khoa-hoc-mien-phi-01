const SoundManager = {
ctx: null,
sfxEnabled: true,
musicEnabled: false,

init() {
if (!this.ctx) this.ctx = new (window.AudioContext || window.webkitAudioContext)();
const sfx = localStorage.getItem('st-sfx');
const music = localStorage.getItem('st-music');
if (sfx !== null) this.sfxEnabled = sfx === 'true';
if (music !== null) this.musicEnabled = music === 'true';
},

_tone(freq, duration, type = 'sine', volume = 0.15) {
if (!this.sfxEnabled || !this.ctx) return;
try { this.ctx.resume(); } catch(e) {}
const osc = this.ctx.createOscillator();
const gain = this.ctx.createGain();
osc.type = type;
osc.frequency.value = freq;
gain.gain.setValueAtTime(volume, this.ctx.currentTime);
gain.gain.exponentialRampToValueAtTime(0.001, this.ctx.currentTime + duration);
osc.connect(gain);
gain.connect(this.ctx.destination);
osc.start();
osc.stop(this.ctx.currentTime + duration);
},

playClick() { this._tone(800, 0.08, 'sine', 0.08); },

playCorrect() {
this._tone(523, 0.15, 'sine', 0.12);
setTimeout(() => this._tone(784, 0.2, 'sine', 0.12), 120);
},

playWrong() {
this._tone(330, 0.15, 'sine', 0.08);
setTimeout(() => this._tone(262, 0.2, 'sine', 0.08), 120);
},

playComplete() {
[523, 659, 784, 1047].forEach((f, i) => {
setTimeout(() => this._tone(f, 0.25, 'sine', 0.1), i * 100);
});
},

playLevelUp() {
[523, 659, 784, 1047, 1319].forEach((f, i) => {
setTimeout(() => this._tone(f, 0.3, 'triangle', 0.12), i * 120);
});
},

playCourseComplete() {
if (!this.sfxEnabled || !this.ctx) return;
try { this.ctx.resume(); } catch(e) {}
// Grand celebration: C major arpeggio up, then triumphant chord, then sparkle
const notes = [523, 659, 784, 1047, 1319, 1568];
notes.forEach((f, i) => { setTimeout(() => this._tone(f, 0.35, 'triangle', 0.12), i * 110); });
// Triumphant chord at the end
setTimeout(() => {
[1047, 1319, 1568].forEach(f => this._tone(f, 0.6, 'sine', 0.08));
}, 750);
// Sparkle high notes
setTimeout(() => { this._tone(2093, 0.15, 'sine', 0.06); }, 1100);
setTimeout(() => { this._tone(2637, 0.15, 'sine', 0.05); }, 1250);
setTimeout(() => { this._tone(3136, 0.2, 'sine', 0.04); }, 1400);
// Final grand chord
setTimeout(() => {
[1047, 1319, 1568, 2093].forEach(f => this._tone(f, 1.0, 'sine', 0.06));
}, 1600);
},

playNotification() { this._tone(1047, 0.1, 'sine', 0.06); },

musicOsc: null,
musicGain: null,

startMusic() {
if (!this.musicEnabled || !this.ctx || this.musicOsc) return;
try { this.ctx.resume(); } catch(e) {}
this.musicGain = this.ctx.createGain();
this.musicGain.gain.value = 0.03;
this.musicGain.connect(this.ctx.destination);
this.musicOsc = [261.63, 329.63, 392].map(freq => {
const osc = this.ctx.createOscillator();
osc.type = 'sine';
osc.frequency.value = freq;
osc.connect(this.musicGain);
osc.start();
return osc;
});
},

stopMusic() {
if (this.musicOsc) {
this.musicOsc.forEach(o => { try { o.stop(); } catch(e) {} });
this.musicOsc = null;
}
},

toggleSfx() {
this.sfxEnabled = !this.sfxEnabled;
localStorage.setItem('st-sfx', this.sfxEnabled);
if (this.sfxEnabled) this.playClick();
return this.sfxEnabled;
},

toggleMusic() {
this.musicEnabled = !this.musicEnabled;
localStorage.setItem('st-music', this.musicEnabled);
if (this.musicEnabled) this.startMusic();
else this.stopMusic();
return this.musicEnabled;
}
};

document.addEventListener('click', function initSound() {

document.removeEventListener('click', initSound);

function SoundToggle() {
const [sfx, setSfx] = useState(() => { const s = localStorage.getItem('st-sfx'); return s === null ? true : s === 'true'; });
const [music, setMusic] = useState(() => { const m = localStorage.getItem('st-music'); return m !== null && m === 'true'; });

return (
<div className="fixed top-3 right-3 z-50 flex gap-1 no-print">
<button onClick={() => setSfx(SoundManager.toggleSfx())}
className="w-8 h-8 rounded-full bg-white/80 backdrop-blur border border-gray-200 flex items-center justify-center text-sm hover:bg-gray-100 transition-all shadow-sm"
title={sfx ? ui().soundOn : ui().soundOff}>
{sfx ? '🔊' : '🔇'}
</button>
<button onClick={() => setMusic(SoundManager.toggleMusic())}
className="w-8 h-8 rounded-full bg-white/80 backdrop-blur border border-gray-200 flex items-center justify-center text-sm hover:bg-gray-100 transition-all shadow-sm"
title={music ? ui().musicOn : ui().musicOff}>
{music ? '🎵' : '🎶'}
</button>
</div>
);
}

function ExerciseIdentifySystem({ exercise, level, onComplete }) {
const [placements, setPlacements] = useState({});
const [dragItem, setDragItem] = useState(null);
const [showResults, setShowResults] = useState(false);

const unplaced = exercise.items.filter(i => !placements[i.id]);
const linearItems = exercise.items.filter(i => placements[i.id] === 'linear');
const systemsItems = exercise.items.filter(i => placements[i.id] === 'systems');

const handleDragStart = (id) => setDragItem(id);
const handleDrop = (category) => {
if (dragItem) {
setPlacements(prev => ({ ...prev, [dragItem]: category }));
setDragItem(null);
}
};
const handleRemove = (id) => {
setPlacements(prev => { const n = {...prev}; delete n[id]; return n; });
};

const allPlaced = Object.keys(placements).length === exercise.items.length;
const score = exercise.items.filter(i => placements[i.id] === i.answer).length;

const checkAnswers = () => {
setShowResults(true);
const pct = score === exercise.items.length ? 100 : Math.round((score / exercise.items.length) * 100);
onComplete(pct);
setTimeout(() => { if (pct >= 60) SoundManager.playComplete(); else SoundManager.playWrong(); }, 200);
};

const renderCard = (item, removable = false) => (
<div key={item.id}
draggable={!showResults}
onDragStart={() => handleDragStart(item.id)}
className={`p-3 rounded-lg text-sm cursor-grab active:cursor-grabbing border transition-all ${
showResults
? placements[item.id] === item.answer
? 'border-green-300 bg-green-50'
: 'border-red-300 bg-red-50'
: 'border-gray-200 bg-white hover:border-brand-300 hover:shadow-sm'
}`}>
<div className="flex items-start justify-between gap-2">
<span>{item.text}</span>
{removable && !showResults && (
<button onClick={() => handleRemove(item.id)} className="text-gray-400 hover:text-red-500 flex-shrink-0">&times;</button>
)}
</div>
{showResults && placements[item.id] !== item.answer && (
<div className="mt-2 text-xs text-red-600 font-medium">{ui().shouldBe} {item.answer === 'systems' ? ui().systemsThinking : ui().linearThinking}</div>
)}
</div>
);

return (
<div className="space-y-6 fade-in">
<div className="bg-amber-50 border border-amber-200 rounded-xl p-4 text-sm text-amber-800">

function ExerciseBuildLoop({ exercise, onComplete }) {
const [currentScenario, setCurrentScenario] = useState(0);
const [selectedOrder, setSelectedOrder] = useState([]);
const [selectedType, setSelectedType] = useState(null);
const [showResult, setShowResult] = useState(false);
const [shuffledIndices] = useState(() => {
return exercise.scenarios.map(s => {
const indices = s.elements.map((_, i) => i);
for (let i = indices.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [indices[i], indices[j]] = [indices[j], indices[i]]; }
return indices;
});
});

const scenario = exercise.scenarios[currentScenario];
const availableIndices = shuffledIndices[currentScenario].filter(i => !selectedOrder.includes(i));

const addToOrder = (idx) => { if (!showResult) setSelectedOrder(prev => [...prev, idx]); };
const removeFromOrder = (idx) => { if (!showResult) setSelectedOrder(prev => prev.filter(i => i !== idx)); };

const checkAnswer = () => {
setShowResult(true);
const orderCorrect = JSON.stringify(selectedOrder) === JSON.stringify(scenario.correctOrder);
const typeCorrect = selectedType === scenario.loopType;
const score = (orderCorrect ? 50 : 0) + (typeCorrect ? 50 : 0);
onComplete(score);
setTimeout(() => { if (score >= 60) SoundManager.playComplete(); else SoundManager.playWrong(); }, 200);
};

return (
<div className="space-y-6 fade-in">
<div className="bg-brand-50 border border-brand-200 rounded-xl p-4 text-sm text-brand-800 font-semibold">
{ui().scenarioLabel} {scenario.name}
</div>
<div>
<h5 className="text-sm font-semibold text-gray-500 uppercase tracking-wide mb-2">{ui().elementsProvided}</h5>
<div className="flex flex-wrap gap-2">
{availableIndices.map(idx => (
<button key={idx} onClick={() => addToOrder(idx)}
className="px-4 py-2 rounded-lg bg-white border border-gray-200 hover:border-brand-400 hover:bg-brand-50 text-sm transition-all">
{scenario.elements[idx]}
</button>
))}
</div>
</div>
<div>
<h5 className="text-sm font-semibold text-gray-500 uppercase tracking-wide mb-2">{ui().yourLoopOrder}</h5>
<div className="min-h-[60px] rounded-xl border-2 border-dashed border-gray-300 p-3 flex flex-wrap items-center gap-2">
{selectedOrder.length === 0 && <span className="text-gray-400 text-sm italic">{ui().clickToBuild}</span>}
{selectedOrder.map((idx, pos) => (
<div key={pos} className="flex items-center gap-1">
<button onClick={() => removeFromOrder(idx)}
className="px-3 py-1.5 rounded-lg bg-brand-100 text-brand-700 text-sm font-medium hover:bg-red-100 hover:text-red-700 transition-colors">
{scenario.elements[idx]}
</button>
{pos < selectedOrder.length - 1 && <span className="text-gray-400 text-lg">&rarr;</span>}
</div>
))}
{selectedOrder.length === scenario.elements.length && <span className="text-gray-400 text-lg">{ui().loopsBack}</span>}
</div>
</div>
{selectedOrder.length === scenario.elements.length && (
<div>
<h5 className="text-sm font-semibold text-gray-500 uppercase tracking-wide mb-2">{ui().loopType}</h5>
<div className="flex gap-3">
{[{ key: 'reinforcing', label: ui().reinforcing, icon: '🔄' }, { key: 'balancing', label: ui().balancing, icon: '⚖️' }].map(lt => (
<button key={lt.key} onClick={() => !showResult && setSelectedType(lt.key)}
className={`flex-1 p-4 rounded-xl border-2 text-center transition-all ${
selectedType === lt.key ? 'border-brand-500 bg-brand-50' : 'border-gray-200 hover:border-gray-300'
}`}>
<div className="text-2xl mb-1">{lt.icon}</div>
<div className="font-semibold text-sm">{lt.label}</div>
</button>
))}
</div>
</div>
)}
{!showResult && (
<Button onClick={checkAnswer} disabled={selectedOrder.length < scenario.elements.length || !selectedType} size="lg" className="w-full">
{ui().checkAnswers}
</Button>
)}
{showResult && (
<div className="rounded-xl border bg-blue-50 border-blue-200 p-5">
<div className="font-bold text-lg mb-2">{selectedType === scenario.loopType ? ui().correctLoopType : ui().wrongLoopType}</div>
<p className="text-sm text-gray-700">{scenario.explanation}</p>
</div>
)}
</div>
);
}

function ExerciseLeverageRanking({ exercise, onComplete }) {

const [dragIdx, setDragIdx] = useState(null);

const newItems = [...items];
const [moved] = newItems.splice(fromIdx, 1);
newItems.splice(toIdx, 0, moved);
setItems(newItems);
};

const check = () => {
setShowResult(true);
const correct = items.every((item, i) => item.level === i + 1);
const score = correct ? 100 : 50;
onComplete(score);
setTimeout(() => { if (score >= 60) SoundManager.playComplete(); else SoundManager.playWrong(); }, 200);
};

<div className="space-y-6 fade-in">
<div className="bg-amber-50 border border-amber-200 rounded-xl p-4 text-sm text-amber-800">

</div>

<div key={item.id}
draggable
onDragStart={() => setDragIdx(idx)}
onDragOver={e => e.preventDefault()}
onDrop={() => { if (dragIdx !== null && dragIdx !== idx) { moveItem(dragIdx, idx); setDragIdx(null); } }}

function Assessment({ assessment, onComplete }) {
const [currentQ, setCurrentQ] = useState(0);
const [answers, setAnswers] = useState({});
const [showExplanation, setShowExplanation] = useState(false);
const [finished, setFinished] = useState(false);

const q = assessment.questions[currentQ];
const selected = answers[currentQ];
const isCorrect = selected === q.correct;
const totalCorrect = Object.entries(answers).filter(([k, v]) => assessment.questions[parseInt(k)].correct === v).length;

const selectAnswer = (idx) => {
if (showExplanation) return;
setAnswers(prev => ({ ...prev, [currentQ]: idx }));
setShowExplanation(true);
setTimeout(() => { if (idx === q.correct) SoundManager.playCorrect(); else SoundManager.playWrong(); }, 100);
};

const nextQuestion = () => {
setShowExplanation(false);
if (currentQ < assessment.questions.length - 1) {
setCurrentQ(currentQ + 1);
} else {
setFinished(true);
const pct = Math.round((totalCorrect / assessment.questions.length) * 100);
onComplete(pct);
setTimeout(() => { if (pct >= 60) SoundManager.playLevelUp(); else SoundManager.playWrong(); }, 200);
}
};

if (finished) {
const pct = Math.round((totalCorrect / assessment.questions.length) * 100);
const passed = pct >= 60;
return (
<div className="text-center space-y-4 fade-in">
{passed && <div className="bounce-in"><CelebrationSVG className="mx-auto mb-2"/></div>}
<div className={`inline-flex items-center justify-center w-24 h-24 rounded-full text-3xl font-bold ${passed ? 'bg-green-100 text-green-700' : 'bg-amber-100 text-amber-700'}`}>
{pct}%
</div>
<h3 className="text-xl font-bold">{passed ? ui().passed : ui().keepLearning}</h3>
<p className="text-gray-600 text-sm">{totalCorrect}/{assessment.questions.length} {ui().correctAnswers}</p>
<p className="text-gray-500 text-sm">{passed ? ui().passedMsg : ui().failedMsg}</p>
</div>
);
}

return (
<div className="space-y-6 fade-in">
<div className="flex items-center justify-between mb-2">
<Badge color="brand">{ui().question} {currentQ + 1}/{assessment.questions.length}</Badge>
<span className="text-sm text-gray-500">{totalCorrect} {ui().correct}</span>
</div>
<ProgressBar value={currentQ + 1} max={assessment.questions.length} />
<div className="bg-white rounded-xl border p-6">
<h4 className="font-bold text-lg mb-4">{q.q}</h4>
<div className="space-y-3">
{q.opts.map((opt, i) => (
<button key={i} onClick={() => selectAnswer(i)}
className={`w-full text-left p-4 rounded-xl border-2 transition-all text-sm ${
showExplanation
? i === q.correct ? 'border-green-400 bg-green-50 font-medium' : i === selected ? 'border-red-400 bg-red-50' : 'border-gray-200 opacity-50'
: i === selected ? 'border-brand-400 bg-brand-50' : 'border-gray-200 hover:border-gray-300 hover:bg-gray-50'
}`}>
<div className="flex items-center gap-3">
<div className={`w-7 h-7 rounded-full border-2 flex items-center justify-center text-xs font-bold flex-shrink-0 ${
showExplanation
? i === q.correct ? 'border-green-500 bg-green-500 text-white' : i === selected ? 'border-red-500 bg-red-500 text-white' : 'border-gray-300 text-gray-400'
: 'border-gray-300 text-gray-500'
}`}>{String.fromCharCode(65 + i)}</div>
<span>{opt}</span>
</div>
</button>
))}
</div>
</div>
{showExplanation && (
<div className={`rounded-xl p-4 ${isCorrect ? 'bg-green-50 border border-green-200' : 'bg-amber-50 border border-amber-200'}`}>
<p className="font-semibold text-sm">{isCorrect ? ui().correctExcl : ui().wrongExcl} {ui().answerIs} {q.opts[q.correct]}</p>
</div>
)}
{showExplanation && (
<Button onClick={nextQuestion} size="lg" className="w-full">
{currentQ < assessment.questions.length - 1 ? ui().nextQ : ui().seeResults}
</Button>
)}
</div>
);
}

// Store registration in a local queue as a backup (in case Wix fails)
try {
const queue = JSON.parse(localStorage.getItem('st-registration-queue') || '[]');
queue.push({ name, email, registeredAt, status: 'pending' });
localStorage.setItem('st-registration-queue', JSON.stringify(queue));
} catch (e) {}
setScreen('welcome');
// Save to Wix CMS with retry
(async () => {
const saveToWix = async (attempt) => {
console.log('[Wix] Attempting to save registration, attempt:', attempt);
const tokenRes = await fetch('https://www.wixapis.com/oauth2/token', {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify({ clientId: '66d5ac9b-d1fb-4a09-a417-72d691aafb4b', grantType: 'anonymous' })
});
if (!tokenRes.ok) throw new Error('Token fetch failed: ' + tokenRes.status);
const tokenData = await tokenRes.json();
const token = tokenData.access_token;
const insertRes = await fetch('https://www.wixapis.com/wix-data/v2/items', {
method: 'POST',
headers: { 'Content-Type': 'application/json', 'Authorization': token },
body: JSON.stringify({
dataCollectionId: 'SystemsThinkingRegistrations',
dataItem: { data: { fullName: name, email: email, level: '', registeredAt: registeredAt } }
})
});
if (!insertRes.ok) {
const errText = await insertRes.text();
throw new Error('Insert failed: ' + insertRes.status + ' ' + errText);
}
const insertData = await insertRes.json();
console.log('[Wix] ✓ Registration saved successfully:', insertData.dataItem?.id);
if (insertData.dataItem?.id) {
localStorage.setItem('st-wix-record-id', insertData.dataItem.id);
// Mark in queue
try {
const queue = JSON.parse(localStorage.getItem('st-registration-queue') || '[]');
const last = queue[queue.length - 1];
if (last && last.email === email) { last.status = 'saved'; last.wixId = insertData.dataItem.id; }
localStorage.setItem('st-registration-queue', JSON.stringify(queue));
} catch (e) {}
}
return true;
};
// Try 3 times with exponential backoff
for (let attempt = 1; attempt <= 3; attempt++) {
try {
await saveToWix(attempt);
return;
} catch (e) {
console.error('[Wix] Attempt ' + attempt + ' failed:', e.message);
if (attempt < 3) await new Promise(r => setTimeout(r, attempt * 1000));
}
}
console.error('[Wix] All retry attempts failed. Registration is safe in localStorage.');
})();
};

const chooseLevel = (s) => {
setLevel(s);
localStorage.setItem('st-level', s);
setScreen('dashboard');
// Update Wix record with level
const recordId = localStorage.getItem('st-wix-record-id');
if (recordId) {
(async () => {
try {
const tokenRes = await fetch('https://www.wixapis.com/oauth2/token', {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify({ clientId: '66d5ac9b-d1fb-4a09-a417-72d691aafb4b', grantType: 'anonymous' })
});
const tokenData = await tokenRes.json();
await fetch('https://www.wixapis.com/wix-data/v2/items/' + recordId, {
method: 'PUT',
headers: { 'Content-Type': 'application/json', 'Authorization': tokenData.access_token },
body: JSON.stringify({
dataCollectionId: 'SystemsThinkingRegistrations',
dataItem: { id: recordId, data: { fullName: userName, email: userEmail, level: s, registeredAt: localStorage.getItem('st-registered-at') || new Date().toISOString() } }
})
});
} catch (e) { console.log('Wix level update skipped:', e.message); }
})();
}
};

const resetCourse = () => {
setLevel(null);
setProgress({});
localStorage.removeItem('st-level');
localStorage.removeItem('st-progress');
localStorage.removeItem('st-user-name');
localStorage.removeItem('st-user-email');
setUserName('');
setUserEmail('');
setRegName('');
setRegEmail('');
setScreen('register');
};

const updateProgress = (moduleId, type, score) => {
setProgress(prev => ({
...prev,
[moduleId]: { ...prev[moduleId], [type]: score },
}));
};

const completedModules = currentModules.filter(m => (progress[m.id]?.assessment || 0) >= 60).length;
const allComplete = completedModules === currentModules.length;
const levelData = level ? currentLevels.find(s => s.id === level) : null;

// REGISTRATION SCREEN
if (screen === 'register') {
const canRegister = regName.trim().length >= 2 && regEmail.trim().includes('@');
return (
<><LanguageSwitcher lang={lang} onChange={changeLang} /><SoundToggle /><div className="min-h-screen fun-bg confetti-bg flex flex-col px-4">
<div className="flex-1 flex items-center justify-center">
<div className="max-w-md w-full slide-up">
<div className="text-center mb-8">
<div className="flex justify-center mb-3"><img src="npv-logo.jpg" alt="NPV Logo" className="h-16 object-contain" /></div>
<h1 className="font-display text-3xl sm:text-4xl font-black text-gray-900 mb-2">{t.courseTitle}</h1>
<p className="text-gray-500">{t.registerTitle}</p>
</div>
<div className="bg-white/80 backdrop-blur rounded-3xl border-2 border-white shadow-lg p-6 sm:p-8 space-y-5">
{/* Honeypot field - hidden from real users, bots will fill it */}
<div style={{position:'absolute',left:'-9999px',top:'-9999px',width:'1px',height:'1px',overflow:'hidden',opacity:0,pointerEvents:'none'}} aria-hidden="true">
<label htmlFor="website_url">Website (leave blank)</label>
<input
type="text"
id="website_url"
name="website_url"
tabIndex="-1"
autoComplete="off"
value={regHoneypot}
onChange={e => setRegHoneypot(e.target.value)}
/>
</div>
<div>
<label className="block text-sm font-bold text-gray-700 mb-1.5">{t.fullName}</label>
<input
type="text"
value={regName}
onChange={e => setRegName(e.target.value)}
placeholder={lang === 'en' ? 'John Doe' : 'Nguyễn Văn A'}
className="w-full rounded-xl border-2 border-gray-200 px-4 py-3 text-sm focus:ring-2 focus:ring-brand-500 focus:border-brand-500 transition-all"
/>
<p className="text-xs text-amber-600 mt-1.5">{t.nameWarning}</p>
</div>
<div>
<label className="block text-sm font-bold text-gray-700 mb-1.5">{t.email}</label>
<input
type="email"
value={regEmail}
onChange={e => setRegEmail(e.target.value)}
placeholder="email@example.com"
className="w-full rounded-xl border-2 border-gray-200 px-4 py-3 text-sm focus:ring-2 focus:ring-brand-500 focus:border-brand-500 transition-all"
/>
<p className="text-xs text-gray-400 mt-1.5">{t.emailNote}</p>
</div>
<Button onClick={handleRegister} disabled={!canRegister} size="lg" className="w-full pulse-glow">
{t.startBtn}
</Button>
<p className="text-xs text-gray-400 text-center">{t.freeNote}</p>
</div>
</div>
</div>
<Footer />
</div></>
);
}

// WELCOME SCREEN
if (screen === 'welcome') {
return (
<><LanguageSwitcher lang={lang} onChange={changeLang} /><SoundToggle /><div className="min-h-screen fun-bg confetti-bg">
<div className="max-w-5xl mx-auto px-4 py-16 sm:py-20 relative">
{/* Floating decorative shapes */}
<div className="hidden sm:block absolute top-20 left-8 float-slow" style={{animationDelay:'0s'}}><DecorativeShape type="circle" color="#6366f1" size={40}/></div>
<div className="hidden sm:block absolute top-40 right-12 float-slow" style={{animationDelay:'1s'}}><DecorativeShape type="triangle" color="#ec4899" size={35}/></div>
<div className="hidden sm:block absolute bottom-40 left-16 float-slow" style={{animationDelay:'2s'}}><DecorativeShape type="diamond" color="#f59e0b" size={30}/></div>
<div className="hidden sm:block absolute bottom-20 right-20 float-slow" style={{animationDelay:'3s'}}><DecorativeShape type="hexagon" color="#10b981" size={36}/></div>
<div className="hidden sm:block absolute top-32 left-1/4 float-slow" style={{animationDelay:'4s'}}><DecorativeShape type="circle" color="#38bdf8" size={24}/></div>
<div className="hidden sm:block absolute top-60 right-1/4 float-slow" style={{animationDelay:'1.5s'}}><DecorativeShape type="triangle" color="#fb7185" size={28}/></div>

<div className="text-center slide-up relative">
{/* NetworkSVG behind hero text */}
<div className="hidden sm:block absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 opacity-[0.07] pointer-events-none">
<NetworkSVG className="w-[400px] h-[400px] spin-slow"/>
</div>
<div className="inline-flex items-center gap-2 px-5 py-2.5 bg-white/80 backdrop-blur border border-brand-200 text-brand-700 rounded-full text-sm font-bold mb-8 shadow-sm">
<span className="float inline-block">🧩</span> {t.welcomeBadge}
</div>
<h1 className="font-display text-5xl sm:text-7xl font-black text-gray-900 mb-2 tracking-tight leading-tight relative z-[1]">
{t.courseTitle}
</h1>
<h2 className="font-display text-4xl sm:text-6xl font-black hero-gradient mb-6 relative z-[1]">
{t.courseSubtitle}
</h2>
<p className="text-lg sm:text-xl text-gray-600 max-w-2xl mx-auto mb-14 leading-relaxed relative z-[1]">
{t.welcomeSub1}<br className="hidden sm:block" />
{t.welcomeSub2}
</p>
<div className="grid sm:grid-cols-3 gap-5 max-w-3xl mx-auto mb-14 relative">
<div className="hidden sm:block absolute -top-4 -left-6 decoration-dots w-full h-full rounded-3xl pointer-events-none"></div>
{[
{ icon: '🎯', title: t.f1Title, desc: t.f1Desc, bg: 'from-brand-50 to-brand-100', border: 'border-brand-200' },
{ icon: '🎮', title: t.f2Title, desc: t.f2Desc, bg: 'from-accent-50 to-accent-100', border: 'border-accent-200' },

].map((f, i) => (

<div className="text-4xl mb-3 float" style={{ animationDelay: `${i * 0.3}s` }}>{f.icon}</div>
<h3 className="font-bold text-gray-800 mb-1">{f.title}</h3>
<p className="text-gray-500 text-sm">{f.desc}</p>
</div>
))}
</div>
<Button onClick={() => setScreen('level')} size="lg" className="pulse-glow text-lg">
{t.letsGo}
</Button>
</div>
</div>
<Footer />
</div></>
);
}

// LEVEL SELECTION
if (screen === 'level') {
return (
<><LanguageSwitcher lang={lang} onChange={changeLang} /><SoundToggle /><div className="min-h-screen fun-bg confetti-bg">
<div className="max-w-3xl mx-auto px-4 py-16">
<div className="text-center mb-10 fade-in">
<div className="text-5xl mb-4 float">👋</div>
<h2 className="font-display text-3xl sm:text-4xl font-black text-gray-900 mb-3">{t.levelTitle}</h2>
<p className="text-gray-500 max-w-lg mx-auto text-lg">{t.levelSub}</p>
</div>
<div className="grid gap-3">
{currentLevels.map((s, i) => (
<button key={s.id} onClick={() => chooseLevel(s.id)}

style={{ animationDelay: `${i * 100}ms` }}>
<div className={`w-16 h-16 rounded-2xl bg-gradient-to-br ${s.color} flex items-center justify-center text-3xl flex-shrink-0 group-hover:scale-110 group-hover:rotate-3 transition-all shadow-md`}>{s.icon}</div>
<div className="flex-1">
<h3 className="font-bold text-gray-900 text-lg mb-0.5 group-hover:text-brand-700 transition-colors">{s.name}</h3>

</div>
<div className="text-gray-300 group-hover:text-brand-400 text-2xl transition-colors">&rarr;</div>
</button>
))}
</div>

</div>

);
}

<div className="max-w-4xl mx-auto px-4 py-8">

<div className="max-w-4xl mx-auto">

<Certificate level={level} progress={progress} userName={userName} lang={lang} />

<div className="bg-white/80 backdrop-blur-lg border-b border-brand-100 sticky top-0 z-10">
<div className="max-w-5xl mx-auto px-4 py-4 flex items-center justify-between">
<div className="flex items-center gap-3">
<h1 className="font-display text-lg font-bold text-gray-900">🧩 {t.courseTitle}</h1>
<Badge color="brand">{levelData?.icon} {levelData?.name}</Badge>
</div>
<div className="flex items-center gap-3">
<div className="text-sm font-semibold text-brand-600">{completedModules}/{currentModules.length} {t.done}</div>
<button onClick={resetCourse} className="text-xs text-gray-400 hover:text-coral-500 transition-colors">{t.reset}</button>
</div>
</div>
<div className="max-w-5xl mx-auto px-4 pb-3">
<ProgressBar value={completedModules} max={currentModules.length} />
</div>
</div>

<div className="max-w-5xl mx-auto px-4 py-8">
{allComplete && (
<div className="bg-gradient-to-r from-brand-600 via-accent-500 to-pop-400 rounded-3xl p-6 text-white mb-8 flex items-center justify-between flex-wrap gap-4 bounce-in shadow-xl">
<div>
<h3 className="text-xl font-bold mb-1">{t.completeTitle}</h3>
<p className="text-white/80 text-sm">{t.completeSub}</p>
</div>
<Button variant="secondary" onClick={() => { SoundManager.playCourseComplete(); setScreen('certificate'); }}>{t.viewCert}</Button>
</div>
)}

<div className="grid gap-4">
{currentModules.map((mod, i) => {
const mp = progress[mod.id] || {};
const isLocked = i > 0 && !((progress[currentModules[i - 1].id]?.assessment || 0) >= 60);
const isComplete = (mp.assessment || 0) >= 60;
const hasStarted = mp.exercise >= 0 || mp.assessment >= 0;

return (
<button key={mod.id} onClick={() => { if (!isLocked) { setActiveModule(mod.id); setScreen('module'); } }}
disabled={isLocked}
className={`bg-white/80 backdrop-blur rounded-2xl border-2 p-5 sm:p-6 text-left transition-all shadow-sm ${isLocked ? 'opacity-40 cursor-not-allowed border-gray-200' : isComplete ? 'border-mint-300 card-hover cursor-pointer animated-border-left animated-border-left-complete' : 'border-white hover:border-brand-300 card-hover cursor-pointer' + (!isLocked && !isComplete ? ' animated-border-left' : '')}`}>
<div className="flex items-start gap-4">
<div className={`w-14 h-14 rounded-2xl flex items-center justify-center text-2xl flex-shrink-0 ${
isComplete ? 'bg-gradient-to-br from-mint-100 to-mint-200' : hasStarted ? 'bg-gradient-to-br from-brand-100 to-accent-100' : isLocked ? 'bg-gray-100' : 'bg-gradient-to-br from-brand-50 to-accent-50'
}`}>
{isLocked ? '🔒' : isComplete ? '✅' : mod.icon}
</div>
<div className="flex-1 min-w-0">
<div className="flex items-center gap-2 flex-wrap mb-1">
<h3 className="font-bold text-gray-900">{t.module} {mod.id}: {mod.title}</h3>
{isComplete && <Badge color="success">{t.complete}</Badge>}
{hasStarted && !isComplete && <Badge color="warn">{t.inProgress}</Badge>}
</div>
<p className="text-gray-500 text-sm mb-3">{mod.subtitle} · {mod.duration}</p>
{hasStarted && (
<div className="flex gap-4 text-xs text-gray-400">
{mp.exercise >= 0 && <span>{t.tabPractice}: {mp.exercise}%</span>}
{mp.assessment >= 0 && <span>{t.tabAssess}: {mp.assessment}%</span>}
</div>
)}
</div>
<div className="text-gray-300 text-xl flex-shrink-0">&rarr;</div>
</div>
</button>
);
})}
</div>
<Footer />
</div>
</div></>
);
}
